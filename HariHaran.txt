#1. count upper and lower case letters in a string
word = "DecaThlOlOAn"

caps_count=0
small_count=0

def caps():
    global word
    global caps_count
    for i in word:
        if i == i.upper():
            caps_count+=1
    return caps_count
print(caps())


def small():
    global word
    global small_count
    for i in word:
        if i == i.lower():
            small_count+=1
    return small_count

print(small())

#5. what is multiple inheritance explain with the student-sports-result example.
#if a child class can inherit its multiple parent classes attributes and properities is known as multiple inheritance

class Student:
    def show1(self):
        print('Student Name: Hari and his Age: 25')

class Sports(Student):
    def show2(self):
        print('Sports interest in Chess')
        
class Result(Sports):
    def show3(self):
        print('Pass')

n=Result()

n.show1()
n.show2()
n.show3()
